## This is how to create table using terminal
>CREATE TABLE `Supplier`( `delivery_id` INT NOT NULL DEFAULT '0', `delivery` DATE NOT NULL DEFAULT '0000-00-00', `supplier_id` INT NOT NULL DEFAULT '0', primary key(`delivery_id`) );
